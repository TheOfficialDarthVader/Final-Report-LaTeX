\pgfplotsset{
	compat=1.9,
	legend entry/.initial=,

		\pgfkeysgetvalue{/pgfplots/legend entry}\tempValue

		\pgfkeysalso{/pgfplots/forget plot}%

		\expandafter\addlegendentry\expandafter{\tempValue}%
		\fi

}
